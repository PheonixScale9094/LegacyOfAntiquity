integrate_branch = {
		category = influence

		require_acceptance = no

		is_visible = {
			ROOT = {
				is_trade_league_leader = yes
				is_in_trade_league_with = FROM
				has_reform = corporation_reform
			}
			FROM = {
				has_reform = branch_office
			}
		}
		
		is_allowed = {
			FROM = {
				has_opinion = {
					who = ROOT
					value = 50
				}
			}
			FROM = { is_at_war = no }
		}
		
		on_accept = {
			vassalize = FROM
		}
		
		ai_will_do = {
			factor = 1
		}
}
form_subsidiary = {
	category = influence

	require_acceptance = no
       
	is_visible = {
		tag = MDR
		FROM = { 
			NOT = { 
				has_country_flag = mondragon_subsidiary
			}
			NOT = {
				is_subject_of = ROOT
			}
		}
	}
	is_allowed = {
		stability = FROM
		treasury = 350
    }
	on_accept = {
		add_treasury = -350
		FROM = {
			country_event = { 
				id = mondragon_events.101
			}
			set_country_flag = recent_subsidiary_development
		}
	}
	ai_will_do = {
		factor = 1
		modifier = {
			factor = 0.1
			trigger = {
				FROM = {
					provinces_on_capital_continent_of = ROOT
				}
			}
		}
		modifier = {
			factor = 3
			trigger = {
				FROM = {
					NOT = { provinces_on_capital_continent_of = ROOT }
				}
			}
		}
		modifier = {
			factor = 10
			trigger = {
				FROM = {
					num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
						value = 4
						OR = {
							superregion = east_indies_superregion
							superregion = india_superregion
						}
						has_port = yes
					}
				}
			}
		}
		factor = 1
		modifier = {
			factor = 0.1
			trigger = {
				is_at_war = yes
			}
		}
		modifier = {
			factor = 0.1
			trigger = {
				NOT = { treasury = 500 }
			}
		}
		modifier = {
			factor = 5
			trigger = {
				treasury = 700
			}
		}
		modifier = {
			factor = 5
			trigger = {
				treasury = 1000
			}
		}
	}
}
